name: Remote Runner Execution
on:
  push:
    branches:
      - 'code/**'
jobs:
  run-code:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Execute
        run: |
          mkdir -p logs
          # Detect language and run
          if [ -f "src/main.py" ]; then
            python3 src/main.py < input/input.txt > logs/output.txt
          elif [ -f "src/Main.java" ]; then
            javac src/Main.java
            java -cp src Main < input/input.txt > logs/output.txt
          fi
      - name: Push Results
        run: |
          git config user.name "RemoteBot"
          git config user.email "bot@remote.com"
          git add logs/output.txt
          git commit -m "Remote output update"
          git push origin HEAD:logs/${{ github.ref_name }} --force

name: Python Remote Runner
on:
  push:
    branches: [ 'main' ]
permissions:
  contents: write
jobs:
  run-python:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Set up Python 3.12
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - name: Execute
        run: |
          mkdir -p logs
          python3 src/main.py < input/input.txt > logs/output.txt 2>&1
      - name: Push Logs
        run: |
          git config user.name "RemoteBot"
          git config user.email "bot@remote.com"
          git add logs/output.txt
          git commit -m "Update Python Logs" || exit 0
          git push origin HEAD:logs --force

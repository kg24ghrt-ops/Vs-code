name: Atomic Execution Workflow v5
on: [push]

jobs:
  execute-run:
    runs-on: ubuntu-latest
    permissions:
      contents: write

    steps:
      - name: Checkout Exact Commit
        uses: actions/checkout@v4
        with:
          fetch-depth: 0
          ref: ${{ github.sha }}

      - name: Setup Python
        uses: actions/setup-python@v4
        with:
          python-version: 3.11

      - name: Setup Environment
        run: |
          mkdir -p logs input
          git config user.name "Remote Bot"
          git config user.email "bot@edu.com"

      - name: Parse Metadata & Execute
        run: |
          RUN_META=$(ls input/run_*.json 2>/dev/null | sort -r | head -n 1)
          if [ -z "$RUN_META" ]; then echo "ERROR: No metadata"; exit 1; fi

          RUN_ID=$(python3 - <<EOF
import json, sys
with open("$RUN_META") as f:
    print(json.load(f).get("runId", "unknown"))
EOF
          )

          LOG_FILE="logs/output_${RUN_ID}.txt"
          echo ">>> JOB STARTED <<<" > "$LOG_FILE"
          set -o pipefail
          python src/main.py < "input/input_${RUN_ID}.txt" >> "$LOG_FILE" 2>&1 || {
            echo -e "\n--- EXECUTION FAILED ---" >> "$LOG_FILE"
            exit 1
          }

          echo -e "\n--- FINISHED ---" >> "$LOG_FILE"

      - name: Sync Logs to Branch
        if: always()
        run: |
          git fetch origin logs || git checkout -b logs
          git switch logs
          ls -t logs/output_*.txt 2>/dev/null | tail -n +21 | xargs rm -f 2>/dev/null || true
          git add logs/
          git commit -m "Results: Run ${RUN_ID}" || echo "No changes to commit"
          git push origin logs --force
